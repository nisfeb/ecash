const API = "/apps/ecash-services/admin/api";

// ── Fetch helper: same-origin cookie, JSON parse, surface non-200 detail ──
async function req(path, opts) {
  opts = opts || {};
  const headers = opts.body ? {'Content-Type': 'application/json', ...(opts.headers || {})} : (opts.headers || {});
  const r = await fetch(API + path, {credentials: 'same-origin', ...opts, headers});
  let data;
  try { data = await r.json(); } catch { data = null; }
  if (!r.ok) {
    const detail = data && (data.detail || data.error || data.message);
    throw new Error(detail || ('HTTP ' + r.status));
  }
  return data;
}
window.req = req;

// ── Misc helpers ─────────────────────────────────────────────
function toast(msg) {
  const t = document.getElementById("toast");
  t.textContent = msg || 'Copied!';
  t.style.opacity = "1";
  setTimeout(() => { t.style.opacity = "0"; }, 2200);
}
window.toast = toast;

function copyText(t) {
  navigator.clipboard.writeText(t).then(() => toast('Copied!'));
}
window.copyText = copyText;

const esc = (s) => String(s == null ? '' : s)
  .replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
  .replace(/'/g, '&#39;').replace(/"/g, '&quot;');

function fmtDate(unix) {
  if (!unix) return '?';
  return new Date(unix * 1000).toLocaleString();
}

function box(val, label) {
  const isLong = String(val).length > 16;
  return '<div class="box"><div class="num ' + (isLong ? 'sm' : '') + '">' + val + '</div><div class="lbl">' + label + '</div></div>';
}

// ── Shared auto-refresh helper ───────────────────────────────
let _ar = null, _arAt = 0;
function startAuto(loader, sec) {
  stopAuto();
  if (!document.getElementById('autorefresh')?.checked) return;
  _ar = setInterval(() => loader(true), (sec || 10) * 1000);
}
function stopAuto() { if (_ar) { clearInterval(_ar); _ar = null; } }
function stampUpdated() {
  _arAt = Date.now();
  const el = document.getElementById('updated'); if (el) el.textContent = 'updated just now';
}
setInterval(() => {
  const el = document.getElementById('updated');
  if (el && _arAt) { const s = Math.round((Date.now() - _arAt) / 1000); el.textContent = `updated ${s}s ago`; }
}, 1000);
const guard = (msg, fn) => { if (confirm(msg)) fn(); };

document.querySelectorAll("#main-tabs .tab").forEach(btn => {
  btn.addEventListener("click", () => {
    const t = btn.dataset.tab;
    document.querySelectorAll("#main-tabs .tab").forEach(x => x.classList.remove("on"));

  });

// ── Overview tab (read-only, auto-refresh) ───────────────────
async function loadOverview(silent) {
  try {
    const [cred, svc] = await Promise.all([req('/cred/overview'), req('/services')]);
    const services = svc.services || [];
    const activeCount = services.filter(s => s.active).length;
    const totalIssued = services.reduce((a, s) => a + (s.issued || 0), 0);
    const totalRedeemed = services.reduce((a, s) => a + (s.redeemed || 0), 0);

    document.getElementById("ov-stats").innerHTML =
      box(activeCount + ' / ' + services.length, "Services active") +
      box(cred.cred_keysets || 0, "Cred keysets") +
      box(cred.cred_issued || 0, "Credentials issued") +
      box(cred.cred_spent || 0, "Credentials spent") +
      box(totalIssued, "Total issued") +
      box(totalRedeemed, "Total redeemed");

    const gEl = document.getElementById("ov-gauges");
    if (!services.length) {
      gEl.innerHTML = '<div class="empty">No services yet.</div>';
    } else {
      gEl.innerHTML = services.map(s => {
        let line, gauge = '';
        if (s.max_issuance != null) {
          const pct = s.max_issuance > 0 ? Math.min(100, Math.round((s.issued / s.max_issuance) * 100)) : 0;
          line = '<b>' + s.issued + ' / ' + s.max_issuance + '</b> (' + pct + '%)';
          gauge = '<div class="gauge"><div class="gauge-fill" style="width:' + pct + '%"></div></div>';
        } else {
          line = 'issued <b>' + s.issued + '</b>';
        }
        return '<div style="margin-bottom:12px">' +
          '<div style="display:flex;justify-content:space-between;font-size:.82rem">' +
            '<span class="accent">' + esc(s.name) + '</span>' +
            '<span class="muted">' + line + '</span>' +
          '</div>' + gauge +
        '</div>';
      }).join('');
    }
    stampUpdated();
  } catch(e) {
    if (!silent) toast(e.message);
  }
}
window.loadOverview = loadOverview;

document.getElementById("autorefresh").addEventListener("change", e => {
  if (e.target.checked) startAuto(loadOverview); else stopAuto();
});

// ── Services tab (CRUD + allowlist) ──────────────────────────
async function loadServices() {
  try {
    const d = await req('/services');
    const services = d.services || [];
    const listEl = document.getElementById("svc-list");
    if (!services.length) {
      listEl.innerHTML = '<div class="empty">No services yet. Use the form above to create one.</div>';
      return;
    }
    const nowSec = Math.floor(Date.now() / 1000);
    listEl.innerHTML = services.map(s => {
      const isExpired = s.expires && s.expires < nowSec;
      const keys = Array.isArray(s.allowlist) ? s.allowlist : [];
      const locked = s.allowlist_required || keys.length > 0;

      const badges = [];
      badges.push(s.active
        ? '<span class="badge b-active">active</span>'
        : '<span class="badge b-inactive">inactive</span>');
      if (isExpired) badges.push('<span class="badge b-expired">expired</span>');
      const keyCount = s.allowlist_count != null ? s.allowlist_count : keys.length;
      badges.push(locked
        ? '<span class="badge b-locked">&#x1F510; ' + keyCount + ' key' + (keyCount === 1 ? '' : 's') + '</span>'
        : '<span class="badge b-public">public</span>');

      const expText = s.expires
        ? (isExpired ? 'expired ' : 'expires ') + fmtDate(s.expires)
        : 'no expiry';

      let capLine, gauge = '';
      if (s.max_issuance != null) {
        const pct = s.max_issuance > 0 ? Math.min(100, Math.round((s.issued / s.max_issuance) * 100)) : 0;
        capLine = '<span>Issued: <b>' + s.issued + ' / ' + s.max_issuance + '</b> (' + pct + '%)</span>';
        gauge = '<div class="gauge"><div class="gauge-fill" style="width:' + pct + '%"></div></div>';
      } else {
        capLine = '<span>Issued: <b>' + s.issued + '</b></span>';
      }

      const nm = esc(s.name);
      const actions = (s.active
        ? '<button class="rbtn js-svc-deact" data-name="' + nm + '">Deactivate</button>'
        : '<button class="rbtn js-svc-act" data-name="' + nm + '">Activate</button>' +
          ' <button class="rbtn js-svc-del" data-name="' + nm + '">Delete</button>') +
        ' <button class="rbtn js-svc-edit" data-name="' + nm + '">Edit</button>';

      const keyRows = keys.map(k =>
        '<div class="allowlist-key">' +
          '<span class="val">' + esc(k) + '</span>' +
          '<button class="k-copy js-key-copy" title="Copy" data-key="' + esc(k) + '">copy</button>' +
          '<button class="k-del js-key-remove" title="Revoke" data-name="' + nm + '" data-key="' + esc(k) + '">&times;</button>' +
        '</div>'
      ).join('');

      const allowlistBox =
        '<div class="allowlist-box">' +
          '<div class="hdr-row">' +
            '<span>Access keys ' + (locked ? '(gated)' : '(empty &rarr; public)') + '</span>' +
          '</div>' +
          keyRows +
          '<div class="allowlist-add">' +
            '<input class="al-input" placeholder="new access key" />' +
            '<button class="rbtn js-key-add" data-name="' + nm + '">Add</button>' +
          '</div>' +
        '</div>';

      const editBox =
        '<div class="svc-edit" style="display:none;margin-top:10px;padding:10px;background:#0b1413;border:1px solid #134e4a;border-radius:8px">' +
          '<div class="frow">' +
            '<div class="fld"><label>Title</label><input class="ed-title" placeholder="title" value="' + esc(s.title == null ? '' : s.title) + '"></div>' +
            '<div class="fld"><label>Description</label><input class="ed-desc" placeholder="description" value="' + esc(s.description == null ? '' : s.description) + '"></div>' +
          '</div>' +
          '<div class="frow">' +
            '<div class="fld"><label>Max issuance</label><input class="ed-max" type="number" min="0" placeholder="unlimited" value="' + (s.max_issuance != null ? s.max_issuance : '') + '"></div>' +
            '<div class="fld"><label>Expires (unix secs)</label><input class="ed-expires" type="number" min="0" placeholder="never" value="' + (s.expires != null ? s.expires : '') + '"></div>' +
            '<button class="btn btn-sm js-svc-save" data-name="' + nm + '">Save</button>' +
          '</div>' +
        '</div>';

      return '<div class="svc-card">' +
        '<div style="display:flex;justify-content:space-between;align-items:flex-start;gap:8px">' +
          '<div>' +
            '<div class="svc-name">' + nm + ' ' + badges.join(' ') + '</div>' +
            '<div class="svc-title">' + esc(s.title) + '</div>' +
            '<div class="svc-desc">' + esc(s.description) + '</div>' +
          '</div>' +
          '<div style="display:flex;gap:6px;flex-wrap:wrap;justify-content:flex-end">' + actions + '</div>' +
        '</div>' +
        '<div class="svc-meta">' +
          capLine +
          '<span>Redeemed: <b>' + s.redeemed + '</b></span>' +
          '<span>Kind: <b>' + esc(s.kind) + '</b></span>' +
          '<span>' + expText + '</span>' +
        '</div>' +
        gauge +
        editBox +
        allowlistBox +
      '</div>';
    }).join('');
    attachSvcDelegation();
  } catch(e) {
    const listEl = document.getElementById("svc-list");
    if (listEl) listEl.innerHTML = '<div class="empty">Error: ' + esc(e.message) + '</div>';
  }
}
window.loadServices = loadServices;

// Toggle the inline edit panel for a service card.
const startEdit = btn => {
  const card = btn.closest('.svc-card');
  const ed = card ? card.querySelector('.svc-edit') : null;
  if (ed) ed.style.display = (ed.style.display === 'none' || !ed.style.display) ? 'block' : 'none';
};

// Collect changed/non-empty fields and POST /services/update.
const saveEdit = btn => {
  const card = btn.closest('.svc-card');
  if (!card) return;
  const patch = {};
  const title = card.querySelector('.ed-title')?.value;
  const desc = card.querySelector('.ed-desc')?.value;
  const maxV = card.querySelector('.ed-max')?.value?.trim();
  const expV = card.querySelector('.ed-expires')?.value?.trim();
  if (title != null && title !== '') patch.title = title;
  if (desc != null && desc !== '') patch.description = desc;
  if (maxV) patch.max_issuance = Number(maxV);
  if (expV) patch.expires = Number(expV);
  updateService(btn.dataset.name, patch);
};

// One delegated listener for the Services panel (attach once).
let _svDelegated = false;
function attachSvcDelegation() {
  if (_svDelegated) return;
  const sv = document.getElementById('sv') || document.getElementById('sec-sv');
  if (!sv) return;
  sv.addEventListener('click', (e) => {
    const t = e.target.closest('button'); if (!t) return;
    const n = t.dataset.name, k = t.dataset.key;
    if (t.classList.contains('js-svc-deact')) deactivateService(n);
    else if (t.classList.contains('js-svc-act')) activateService(n);
    else if (t.classList.contains('js-svc-del')) deleteService(n);
    else if (t.classList.contains('js-key-remove')) removeKey(n, k);
    else if (t.classList.contains('js-key-copy')) copyText(k);
    else if (t.classList.contains('js-key-add')) addKeyForCard(t);
    else if (t.classList.contains('js-svc-edit')) startEdit(t);
    else if (t.classList.contains('js-svc-save')) saveEdit(t);
  });
  _svDelegated = true;
}

// ── Credentials tab (raw cred keysets) ───────────────────────
async function loadCreds() {
  try {
    const d = await req('/cred/overview');
    document.getElementById("cr-stats").innerHTML =
      box(d.cred_keysets || 0, "Cred keysets") +
      box(d.cred_issued || 0, "Credentials issued") +
      box(d.cred_spent || 0, "Credentials spent");

    const keysets = d.keysets || [];
    const listEl = document.getElementById("cr-list");
    if (!keysets.length) {
      listEl.innerHTML = '<div class="empty">No credential keysets. Use Generate to create one.</div>';
      return;
    }
    listEl.innerHTML =
      '<table class="tbl"><thead><tr><th>Keyset ID</th><th>Status</th><th>Actions</th></tr></thead><tbody>' +
      keysets.map(k => {
        const id = esc(k.id);
        return '<tr>' +
          '<td class="mono accent js-cred-copy" style="cursor:pointer" data-id="' + id + '" title="Click to copy">' + id + '</td>' +
          '<td>' + (k.active ? '<span class="badge b-active">active</span>' : '<span class="badge b-inactive">inactive</span>') + '</td>' +
          '<td style="white-space:nowrap">' +
            (k.active
              ? '<button class="rbtn js-cred-deact" data-id="' + id + '">Deactivate</button>'
              : '<button class="rbtn js-cred-act" data-id="' + id + '">Activate</button>') +
          '</td>' +
        '</tr>';
      }).join('') +
      '</tbody></table>';
    attachCredDelegation();
  } catch(e) {
    document.getElementById("cr-list").innerHTML = '<div class="empty">Error: ' + esc(e.message) + '</div>';
  }
}
window.loadCreds = loadCreds;

// One delegated listener for the Credentials panel (attach once).
let _crDelegated = false;
function attachCredDelegation() {
  if (_crDelegated) return;
  const cr = document.getElementById('cr') || document.getElementById('sec-cr');
  if (!cr) return;
  cr.addEventListener('click', (e) => {
    const t = e.target.closest('.js-cred-copy, button'); if (!t) return;
    const id = t.dataset.id; if (id == null) return;
    if (t.classList.contains('js-cred-act')) actCredKs(id);
    else if (t.classList.contains('js-cred-deact')) deactCredKs(id);
    else if (t.classList.contains('js-cred-copy')) copyText(id);
  });
  _crDelegated = true;
}

const genCredKs   = ()   => req('/cred/keysets/generate',   {method:'POST'}).then(loadCreds).catch(e=>toast(e.message));
const actCredKs   = id   => req('/cred/keysets/activate',   {method:'POST', body: JSON.stringify({id})}).then(loadCreds).catch(e=>toast(e.message));
const deactCredKs = id   => req('/cred/keysets/deactivate', {method:'POST', body: JSON.stringify({id})}).then(loadCreds).catch(e=>toast(e.message));
window.genCredKs = genCredKs;
window.actCredKs = actCredKs;
window.deactCredKs = deactCredKs;
