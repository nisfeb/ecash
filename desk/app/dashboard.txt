const ADMIN = "/apps/ecash/admin/api";
const PUB = "/apps/ecash";
let allQuotes = [];

// ── Helpers ──────────────────────────────────────────────────
async function api(base, path, opts) {
  const r = await fetch(base + path, opts);
  if (!r.ok) {
    let detail;
    try { detail = await r.json(); } catch { detail = await r.text(); }
    throw new Error(typeof detail === 'object' ? JSON.stringify(detail) : detail || "HTTP " + r.status);
  }
  return r.json();
}
const adminApi = (p, o) => api(ADMIN, p, o);
const pubApi = (p, o) => api(PUB, p, o);

const esc = (s) => String(s == null ? '' : s)
  .replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
  .replace(/'/g, '&#39;').replace(/"/g, '&quot;');

function box(val, label, cls) {
  const isLong = String(val).length > 16;
  return `<div class="box"><div class="num ${isLong ? 'sm' : ''} ${cls || ''}">${esc(val)}</div><div class="lbl">${esc(label)}</div></div>`;
}

function badge(text, type) {
  return `<span class="badge b-${type}">${esc(text)}</span>`;
}

function timeAgo(unix) {
  if (!unix) return '?';
  const secs = Math.floor(Date.now() / 1000) - unix;
  if (secs < 60) return secs + 's ago';
  if (secs < 3600) return Math.floor(secs / 60) + 'm ago';
  if (secs < 86400) return Math.floor(secs / 3600) + 'h ago';
  return Math.floor(secs / 86400) + 'd ago';
}

function fmtDate(unix) {
  if (!unix) return '?';
  return new Date(unix * 1000).toLocaleString();
}

function truncId(id, n) {
  if (!id) return '?';
  n = n || 16;
  return id.length > n ? id.slice(0, n) + '...' : id;
}

function copyText(t) {
  navigator.clipboard.writeText(t).then(() => {
    const toast = document.getElementById("toast");
    toast.style.opacity = "1";
    setTimeout(() => { toast.style.opacity = "0"; }, 1500);
  });
}

function showResult(elId, text, type) {
  const el = document.getElementById(elId);
  el.style.display = 'block';
  el.className = 'res' + (type ? ' ' + type : '');
  el.textContent = text;
}

// ── Tab Switching ────────────────────────────────────────────
document.querySelectorAll("#main-tabs .tab").forEach(btn => {
  btn.addEventListener("click", () => {
    document.querySelectorAll("#main-tabs .tab").forEach(t => t.classList.remove("on"));
    document.querySelectorAll(".sec").forEach(s => s.classList.remove("on"));
    btn.classList.add("on");
    document.getElementById("sec-" + btn.dataset.tab).classList.add("on");
    if (btn.dataset.tab === 'ov') { loadOverview(); startAuto(loadOverview); }
    else stopAuto();
  });
});

// ── Stats Bar ────────────────────────────────────────────────
async function loadStats() {
  try {
    const d = await adminApi("/overview");
    const statsEl = document.getElementById("stats");
    const outstanding = (d.total_issued_sats || 0) - (d.total_redeemed_sats || 0);
    statsEl.innerHTML =
      box(d.tokens_issued || 0, "Tokens Issued") +
      box(d.spent_count || 0, "Tokens Spent") +
      box((d.total_issued_sats || 0).toLocaleString(), "Issued (sats)") +
      box((d.total_redeemed_sats || 0).toLocaleString(), "Redeemed (sats)") +
      box(outstanding.toLocaleString(), "Outstanding (sats)") +
      box(d.ln_backend?.configured ? d.ln_backend.type : "none", "LN Backend") +
      box(d.pending_requests || 0, "Pending Requests");

    document.getElementById("sta").textContent = "Online";
    document.getElementById("sta").style.color = "#34d399";
  } catch(e) {
    document.getElementById("sta").textContent = "Error: " + e.message;
    document.getElementById("sta").style.color = "#fca5a5";
  }
}

  stopAuto();
  if (!document.getElementById('autorefresh')?.checked) return;
  _ar = setInterval(() => loader(true), (sec || 10) * 1000);
}
function stopAuto() { if (_ar) { clearInterval(_ar); _ar = null; } }
function stampUpdated() {

  const el = document.getElementById('updated'); if (el) el.textContent = 'updated just now';
}
setInterval(() => {
  const el = document.getElementById('updated');
  if (el && _arAt) { const s = Math.round((Date.now() - _arAt) / 1000); el.textContent = `updated ${s}s ago`; }
}, 1000);

function renderQuotes() {
  let filtered = allQuotes;
  if (currentFilter === 'mint') filtered = allQuotes.filter(q => q._type === 'mint');
  else if (currentFilter === 'melt') filtered = allQuotes.filter(q => q._type === 'melt');
  else if (['UNPAID','PAID','ISSUED'].includes(currentFilter))
    filtered = allQuotes.filter(q => q.state === currentFilter);

  if (!filtered.length) {
    document.getElementById("qt-body").innerHTML = '<div class="empty">No quotes match filter</div>';
    return;
  }

  document.getElementById("qt-body").innerHTML = `
    <table class="tbl">
      <thead><tr>
        <th>Type</th><th>Quote ID</th><th>Amount</th><th>State</th><th>Method</th><th>Created</th><th>Expired</th><th></th>
      </tr></thead>
      <tbody>
        ${filtered.map(q => {
          const stClass = q.state === 'PAID' ? 'paid' : q.state === 'ISSUED' ? 'issued' : q.state === 'PENDING' ? 'pending' : 'unpaid';
          const canDelete = !(q._type === 'mint' && q.state === 'ISSUED') && !(q._type === 'melt' && q.state === 'PAID');
          return `<tr>
            <td>${badge(q._type, q._type === 'mint' ? 'active' : 'pending')}</td>
            <td class="mono" style="cursor:pointer" data-act="copyText" data-arg="${q.quote}" title="${q.quote}">${truncId(q.quote, 16)}</td>
            <td><strong>${q.amount || '?'}</strong> sats</td>
            <td>${badge(q.state, stClass)}</td>
            <td>${q.method || '?'}</td>
            <td class="muted">${fmtDate(q.created)}</td>
            <td>${q.expired ? badge('Yes', 'expired') : '<span class="muted">No</span>'}</td>
            <td>${canDelete ? `<button class="rbtn" style="color:#fca5a5;border-color:#7f1d1d" data-act="deleteQuote" data-arg="${q.quote}" data-arg2="${q._type}">Delete</button>` : ''}</td>
          </tr>`;
        }).join("")}
      </tbody>
    </table>
    <div class="hint" style="margin-top:8px">Showing ${filtered.length} of ${allQuotes.length} quotes. Click quote ID to copy.</div>
  `;
}

// Event delegation: CSP blocks inline on*= handlers, so elements carry data-act
// (+ optional data-arg/data-arg2) and we resolve them to functions here. Args are
// data, never code — strictly safer than the old inline onclick handlers.
const ACTIONS = {
  loadOverview, updateSettings, generateKeyset, loadKeysets, showKeysetList,
  loadQuotes, loadSpent, checkSpent, loadLightning, toggleLnFields, configureLn,
  testLn, loadInfo, updateInfo, clearLn, copyText, viewKeyset, activateKeyset,
  setKeysetFee, deleteQuote,
};
document.addEventListener('click', (e) => {
  const el = e.target.closest('[data-act]');
  if (!el) return;
  const fn = ACTIONS[el.dataset.act];
  if (!fn) return;
  if ('arg2' in el.dataset) fn(el.dataset.arg, el.dataset.arg2);
  else if ('arg' in el.dataset) fn(el.dataset.arg);
  else fn();
});
document.addEventListener('change', (e) => {
  const el = e.target.closest('[data-change]');
  if (!el) return;
  const fn = ACTIONS[el.dataset.change];
  if (fn) fn();
});
